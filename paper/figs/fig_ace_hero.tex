\definecolor{aceSlateDk}{RGB}{71,85,105}
\definecolor{aceLeaf}{RGB}{198,246,213}
\definecolor{aceLeafDk}{RGB}{34,134,58}
\definecolor{aceRose}{RGB}{255,214,224}
\definecolor{aceRoseDk}{RGB}{192,16,72}
\definecolor{aceAmber}{RGB}{254,215,170}
\definecolor{aceAmberDk}{RGB}{194,65,12}
\definecolor{aceGold}{RGB}{254,243,189}
\definecolor{aceGoldDk}{RGB}{161,98,7}
\definecolor{aceRuby}{RGB}{254,205,211}

\begin{tikzpicture}[
    scale=0.92, transform shape,
    every node/.style={font=\scriptsize},
    state/.style={rectangle, draw=aceSlateDk, rounded corners=2pt,
                  fill=aceSlate, line width=0.6pt, align=center,
                  minimum width=1.25cm, minimum height=0.55cm},
    stepbox/.style={rectangle, draw=aceLeafDk, rounded corners=2pt,
                    fill=aceLeaf, line width=0.7pt, align=center,
                    minimum width=1.55cm, minimum height=0.55cm},
    block/.style={rectangle, draw=#1, rounded corners=3pt, line width=1.0pt,
                  align=center, font=\scriptsize, inner sep=4pt},
    bigblock/.style={block=#1, minimum width=2.65cm, minimum height=1.6cm},
    cand/.style={rectangle, draw=aceRoseDk, rounded corners=2pt,
                 fill=aceRose, line width=0.5pt, align=center,
                 minimum width=1.55cm, minimum height=0.42cm,
                 font=\tiny},
    arrh/.style={-{Stealth[length=2.0mm,width=1.6mm]}, line width=0.9pt,
                 line cap=round},
    arrl/.style={-{Stealth[length=1.7mm,width=1.4mm]}, line width=0.55pt,
                 line cap=round, color=aceMuted},
    fb/.style={-{Stealth[length=2.0mm,width=1.6mm]}, line width=0.9pt,
               dashed, color=aceVioDk, line cap=round},
    feed/.style={-{Stealth[length=1.8mm,width=1.5mm]}, line width=0.7pt,
                 dashed, color=aceMuted, line cap=round},
    note/.style={font=\tiny, color=aceMuted, align=center, inner sep=1pt},
]

% =====================================================================
% TIER 1 -- Outer loop strip (top)
% =====================================================================
\node[anchor=west, font=\bfseries\small] at (-7.2, 4.55) {ACE};

\node[state] (s0)  at (-3.20, 4.55) {$s_t$};
\node[stepbox] (st0) at (-1.45, 4.55) {ACE step};
\node[state]   (s1)  at ( 0.30, 4.55) {$s_{t+1}$};
\node[stepbox] (st1) at ( 2.05, 4.55) {ACE step};
\node[state] (s2)  at ( 3.80, 4.55) {$s_{t+2}$};
\node[font=\large, color=aceMuted] at (4.85, 4.55) {$\cdots$};
\draw[arrl] (s0)  -- (st0);
\draw[arrl] (st0) -- (s1);
\draw[arrl] (s1)  -- (st1);
\draw[arrl] (st1) -- (s2);

\node[note, anchor=west] at (-3.20, 3.85)
  {outer loop: state evolves as the policy executes interventions};

% =====================================================================
% TIER 2 -- Inside one ACE step (dashed container)
% =====================================================================

% Container (drawn first so blocks render on top).
\draw[dashed, color=aceMuted, line width=0.5pt, rounded corners=4pt]
      (-7.30, 3.30) rectangle (7.55, -3.55);
\node[anchor=north west, font=\bfseries\small]
      at (-7.20, 3.18) {Inside one ACE step};

% --- (1) State block --------------------------------------------------
\node[bigblock=aceSlateDk, fill=aceSlate, anchor=west,
      minimum width=2.30cm, minimum height=1.85cm,
      align=left, font=\scriptsize, inner sep=5pt]
      (state) at (-7.10, 1.30) {%
  \textbf{State} $s_t$\\[2pt]
  \scriptsize $\bullet$ graph $\mathcal{G}$\\
  \scriptsize $\bullet$ losses $\{L_i\}$\\
  \scriptsize $\bullet$ history};

% --- (2) Policy block -------------------------------------------------
\node[bigblock=aceLeafDk, fill=aceLeaf, line width=1.4pt,
      minimum width=2.65cm, minimum height=1.85cm]
      (pol) at (-2.05, 1.30) {%
  \textbf{Policy} $\pi_\phi$\\[2pt]
  Qwen2.5-1.5B (LM)\\
  + DPO fine-tuning\\[2pt]
  \scriptsize generates $K{=}4$\\
  \scriptsize candidates};

\draw[arrh] (state.east) -- node[note, above]{prompt} (pol.west);

% --- (3) K=4 Candidates ----------------------------------------------
\node[cand] (c1) at (1.25,  2.00) {$c_1$:\, $\mathrm{do}(X{=}0)$};
\node[cand] (c2) at (1.25,  1.50) {$c_2$:\, $\mathrm{do}(X{=}1)$};
\node[cand] (c3) at (1.25,  1.00) {$c_3$:\, $\mathrm{do}(X{=}2)$};
\node[cand] (c4) at (1.25,  0.50) {$c_4$:\, $\mathrm{do}(X{=}3)$};

\draw[arrh] (pol.east) -- node[note, above]{$K{=}4$}
            (c1.west);
\draw[arrh] (pol.east) -- (c2.west);
\draw[arrh] (pol.east) -- (c3.west);
\draw[arrh] (pol.east) -- (c4.west);

% --- (4) Lookahead block ---------------------------------------------
\node[bigblock=aceAmberDk, fill=aceAmber, line width=1.4pt,
      minimum width=3.00cm, minimum height=1.85cm,
      align=center, font=\scriptsize]
      (look) at (4.40, 1.30) {%
  \textbf{Lookahead}\\[2pt]
  \scriptsize for each $c_k$: clone\\
  \scriptsize learner, simulate,\\[1pt]
  $\Delta\mathcal{L}(c_k){=}L_{\mathrm{bef}}{-}L_{\mathrm{aft}}$\\[2pt]
  \scriptsize $R{=}\Delta\mathcal{L}{+}\alpha w{+}\gamma D$};

\draw[arrh] (c1.east) -- (look.west);
\draw[arrh] (c4.east) -- (look.west);

% --- (5) Best / Worst (right side) -----------------------------------
\node[block=aceGoldDk, fill=aceGold, line width=1.4pt,
      minimum width=1.30cm, minimum height=0.55cm]
      (best) at (6.85, 1.85) {\textbf{best} $c^{*}$};
\node[block=aceRubyDk, fill=aceRuby, line width=0.9pt,
      minimum width=1.30cm, minimum height=0.55cm]
      (worst) at (6.85, 0.75) {\textbf{worst} $c^{-}$};

\draw[arrh] (look.east) -- node[note, sloped, above]{argmax $R$} (best.west);
\draw[arrh] (look.east) -- node[note, sloped, below]{argmin $R$} (worst.west);

% --- (6) Environment + Learner (far right column) --------------------
\node[block=aceSkyDk, fill=aceSky, line width=1.4pt,
      minimum width=1.95cm, minimum height=0.95cm,
      align=center]
      (env) at (6.20, -0.85) {%
  \textbf{Environment} $M^{*}$\\[1pt]
  \scriptsize ground-truth SCM};
\node[block=aceGoldDk, fill=aceGold, line width=1.4pt,
      minimum width=1.95cm, minimum height=0.95cm,
      align=center]
      (learn) at (6.20, -2.55) {%
  \textbf{Learner} $M_\theta$\\[1pt]
  \scriptsize student SCM};

% best -> env (right-angle around): down then right
\draw[arrh, color=aceGoldDk]
      (best.south) |- (env.east)
      node[note, pos=0.35, right=1pt, color=aceGoldDk]
        {execute $c^{*}$};
\draw[arrh] (env.south) -- node[note, right]{data $\mathcal{D}$} (learn.north);

% --- (7) DPO Update (bottom-center) ----------------------------------
\node[block=aceVioDk, fill=aceVio, line width=1.4pt,
      minimum width=4.50cm, minimum height=1.45cm,
      align=center, font=\scriptsize]
      (dpo) at (0.70, -2.10) {%
  \textbf{DPO Update}\\[2pt]
  preference pair $(c^{*},\, c^{-})$\\[1pt]
  $\mathcal{L}_{\mathrm{DPO}}=-\log\sigma\!\big(\beta\,\log\tfrac{\pi_\phi}{\pi_{\mathrm{ref}}}\big)$\\
  \scriptsize $\nabla_\phi$ updates the policy};

% best -> DPO (down-then-left)
\draw[fb] ($(best.south)+(-0.18,0)$) |- ($(dpo.north)+(0.6,0.0)$);
\draw[fb] (worst.south) |- ($(dpo.north)+(-0.6,0.0)$);

% DPO -> Policy (dashed feedback, routed below state via container floor)
\draw[fb] (dpo.west) -- ($(dpo.west)+(-1.3,0)$)
      |- ($(pol.south)+(0,-0.05)$);
\node[note, color=aceVioDk, anchor=west]
      at (-2.05, -2.10) {policy gradient};

% Learner -> State (dashed feedback closing the data loop)
\draw[feed] (learn.west) -- ($(learn.west)+(-1.0,0)$)
      |- ($(state.south)+(0,-0.05)$);
\node[note, anchor=east] at (5.70, -3.15) {next step (state advances)};

\end{tikzpicture}
