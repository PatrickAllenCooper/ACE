% =============================================================================
% Hero figure for ACE paper. Two-tier layout (LatentMAS-style):
%   Top:    outer-loop summary (state -> ACE step -> state' -> ...)
%   Bottom: inside one ACE step (dashed container with full mechanism)
%
% Required preamble in main document:
%   \usepackage{tikz}
%   \usetikzlibrary{arrows.meta, positioning, calc, fit, shapes.geometric,
%                   backgrounds, decorations.pathreplacing}
%
% Layout topology (left to right within the dashed container):
%   State -> Policy -> Candidates -> Lookahead -> (Best on top, Worst below)
%                                                  -> (Env on top, Learner below)
%   DPO Update sits on a second row below, fed by Best/Worst, feeding back to Policy.
%   Learner feeds back to State via a dashed "next step" arrow.
% =============================================================================

\definecolor{aceAmber}{RGB}{254,215,170}

\definecolor{aceGoldDk}{RGB}{161,98,7}

\begin{tikzpicture}[

    every node/.style={font=\scriptsize},
    state/.style={rectangle, draw=aceSlateDk, rounded corners=2pt,
                  fill=aceSlate, line width=0.6pt, align=center,

    stepbox/.style={rectangle, draw=aceLeafDk, rounded corners=2pt,

    block/.style={rectangle, draw=#1, rounded corners=3pt, line width=1.0pt,
                  align=center, font=\scriptsize, inner sep=4pt},
    cand/.style={rectangle, draw=aceRoseDk, rounded corners=2pt,
                 fill=aceRose, line width=0.5pt, align=center,
                 minimum width=1.45cm, minimum height=0.42cm,
                 font=\tiny},
    arrh/.style={-{Stealth[length=2.0mm,width=1.6mm]}, line width=0.9pt,
                 line cap=round},
    arrl/.style={-{Stealth[length=1.7mm,width=1.4mm]}, line width=0.55pt,
                 line cap=round, color=aceMuted},
    fb/.style={-{Stealth[length=2.0mm,width=1.6mm]}, line width=0.9pt,
               dashed, color=aceVioDk, line cap=round},
    feed/.style={-{Stealth[length=1.8mm,width=1.5mm]}, line width=0.7pt,
                 dashed, color=aceMuted, line cap=round},
    note/.style={font=\tiny, color=aceMuted, align=center, inner sep=1pt},
]

% =====================================================================
% TIER 1 -- Outer loop strip (top)
% =====================================================================
\node[anchor=west, font=\bfseries\small] at (-9.0, 4.6) {ACE};

\node[state]   (s0)  at (-4.30, 4.6) {$s_t$};
\node[stepbox] (st0) at (-2.50, 4.6) {ACE step};
\node[state]   (s1)  at (-0.70, 4.6) {$s_{t+1}$};
\node[stepbox] (st1) at ( 1.10, 4.6) {ACE step};
\node[state]   (s2)  at ( 2.90, 4.6) {$s_{t+2}$};
\node[font=\large, color=aceMuted] at (4.10, 4.6) {$\cdots$};
\draw[arrl] (s0)  -- (st0);
\draw[arrl] (st0) -- (s1);
\draw[arrl] (s1)  -- (st1);
\draw[arrl] (st1) -- (s2);

\node[note, anchor=west] at (-4.30, 3.85)
  {outer loop: state evolves as the policy executes interventions};

% =====================================================================
% TIER 2 -- Inside one ACE step (dashed container)
% =====================================================================

% Container.
\draw[dashed, color=aceMuted, line width=0.5pt, rounded corners=4pt]
      (-9.20, 3.30) rectangle (9.30, -4.20);
\node[anchor=north west, font=\bfseries\small]
      at (-9.10, 3.18) {Inside one ACE step};

% --- (1) State block --------------------------------------------------
\node[block=aceSlateDk, fill=aceSlate, anchor=center,
      minimum width=2.20cm, minimum height=1.85cm,
      align=left, font=\scriptsize, inner sep=5pt]
      (state) at (-7.85, 1.50) {%
  \textbf{State} $s_t$\\[2pt]
  \scriptsize $\bullet$ graph $\mathcal{G}$\\
  \scriptsize $\bullet$ losses $\{L_i\}$\\
  \scriptsize $\bullet$ history};

% --- (2) Policy block -------------------------------------------------
\node[block=aceLeafDk, fill=aceLeaf, line width=1.4pt, anchor=center,
      minimum width=2.55cm, minimum height=1.85cm]
      (pol) at (-4.45, 1.50) {%
  \textbf{Policy} $\pi_\phi$\\[2pt]
  Qwen2.5-1.5B (LM)\\
  + DPO fine-tuning\\[2pt]
  \scriptsize generates $K{=}4$\\
  \scriptsize candidates};

\draw[arrh] (state.east) -- node[note, above]{prompt} (pol.west);

% --- (3) K=4 Candidates ----------------------------------------------
\node[cand] (c1) at (-1.55,  2.30) {$c_1$:\, $\mathrm{do}(X{=}0)$};
\node[cand] (c2) at (-1.55,  1.78) {$c_2$:\, $\mathrm{do}(X{=}1)$};
\node[cand] (c3) at (-1.55,  1.26) {$c_3$:\, $\mathrm{do}(X{=}2)$};
\node[cand] (c4) at (-1.55,  0.74) {$c_4$:\, $\mathrm{do}(X{=}3)$};

% Single labelled feeder, then fan-in lines (no arrowhead) for the rest.
\draw[arrh] (pol.east) -- node[note, above, pos=0.5]{$K{=}4$} (c1.west);
\draw[line width=0.55pt, color=aceMuted] (pol.east) -- (c2.west);
\draw[line width=0.55pt, color=aceMuted] (pol.east) -- (c3.west);
\draw[line width=0.55pt, color=aceMuted] (pol.east) -- (c4.west);

% --- (4) Lookahead block ---------------------------------------------
\node[block=aceAmberDk, fill=aceAmber, line width=1.4pt, anchor=center,
      minimum width=2.85cm, minimum height=1.85cm,
      align=center, font=\scriptsize]
      (look) at (1.65, 1.50) {%
  \textbf{Lookahead}\\[2pt]
  \scriptsize for each $c_k$: clone\\
  \scriptsize learner, simulate,\\[1pt]
  $\Delta\mathcal{L}(c_k){=}L_{\mathrm{bef}}{-}L_{\mathrm{aft}}$\\[2pt]
  \scriptsize $R{=}\Delta\mathcal{L}{+}\alpha w{+}\gamma D$};

\draw[arrh] (c1.east) -- (look.west);
\draw[line width=0.55pt, color=aceMuted] (c2.east) -- (look.west);
\draw[line width=0.55pt, color=aceMuted] (c3.east) -- (look.west);
\draw[line width=0.55pt, color=aceMuted] (c4.east) -- (look.west);

% --- (5) Best / Worst (right of Lookahead) ---------------------------
\node[block=aceGoldDk, fill=aceGold, line width=1.4pt,
      minimum width=1.40cm, minimum height=0.62cm]
      (best) at (4.55, 2.10) {\textbf{best} $c^{*}$};
\node[block=aceRubyDk, fill=aceRuby, line width=0.9pt,
      minimum width=1.40cm, minimum height=0.62cm]
      (worst) at (4.55, 0.90) {\textbf{worst} $c^{-}$};

\draw[arrh] (look.east) -- node[note, above, pos=0.5]{argmax $R$}
            ($(best.west)+(0,0.0)$);
\draw[arrh] (look.east) -- node[note, below, pos=0.5]{argmin $R$}
            ($(worst.west)+(0,0.0)$);

% --- (6) Environment + Learner (right of Best/Worst) -----------------
\node[block=aceSkyDk, fill=aceSky, line width=1.4pt,
      minimum width=1.95cm, minimum height=0.95cm,
      align=center]
      (env) at (7.65, 2.10) {%
  \textbf{Environment} $M^{*}$\\[1pt]
  \scriptsize ground-truth SCM};
\node[block=aceTealDk, fill=aceTeal, line width=1.4pt,
      minimum width=1.95cm, minimum height=0.95cm,
      align=center]
      (learn) at (7.65, 0.55) {%
  \textbf{Learner} $M_\theta$\\[1pt]
  \scriptsize student SCM};

% best -> env (clean horizontal arrow, no overlaps)
\draw[arrh, color=aceGoldDk]
      (best.east) -- node[note, above, color=aceGoldDk]{execute $c^{*}$}
      (env.west);
% env -> learn (vertical, "data D" label to the right)
\draw[arrh] (env.south) -- node[note, right]{data $\mathcal{D}$} (learn.north);

% --- (7) DPO Update (bottom-center) ----------------------------------
\node[block=aceVioDk, fill=aceVio, line width=1.4pt,
      minimum width=4.80cm, minimum height=1.35cm,
      align=center, font=\scriptsize]
      (dpo) at (0.50, -2.45) {%
  \textbf{DPO Update}\\[2pt]
  preference pair $(c^{*},\, c^{-})$\\[1pt]
  $\mathcal{L}_{\mathrm{DPO}}=-\log\sigma\!\big(\beta\,\log\tfrac{\pi_\phi}{\pi_{\mathrm{ref}}}\big)$};

% best -> DPO (right-angle: down, then left into DPO top-right)
\draw[fb] (best.south) -- (4.55, -1.30)
          -- ($(dpo.north)+(1.50,0)+(0,0.50)$)
          -- ($(dpo.north)+(1.50,0)$);
% worst -> DPO (right-angle: down, then left into DPO top-left)
\draw[fb] (worst.south) -- (4.55, -0.20)
          -- ($(dpo.north)+(-1.50,0)+(0,0.50)$)
          -- ($(dpo.north)+(-1.50,0)$);

% DPO -> Policy (dashed feedback: left from DPO, up alongside state, right into policy)
\draw[fb] (dpo.west) -- (-6.20, -2.45)
          -- (-6.20, 0.30)
          -- (pol.south |- {0,0.30})
          -- (pol.south);
\node[note, color=aceVioDk, anchor=south, fill=white, inner sep=1pt]
      at (-5.30, 0.32) {policy gradient};

% Learner -> State (long dashed feedback along the container floor)
\draw[feed]
      (learn.south) -- (7.65, -3.65)
      -- (-7.85, -3.65)
      -- (state.south);
\node[note, color=aceMuted, fill=white, inner sep=2pt]
      at (0.00, -3.65) {next step (state advances)};

\end{tikzpicture}
